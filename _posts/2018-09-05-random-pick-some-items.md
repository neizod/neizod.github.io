``` python
def pick_3(n, k):
    remain = list(range(n))
    result = set()
    for i in range(k):
        x = randint(0, n-1-i)
```

``` python
def pick_4(n, k):
    remain = dict()
    result = set()
    for i in range(k):
        x = randint(0, n-1-i)
        if n-1-i not in remain:
            remain[n-1-i] = n-1-i
        if x not in remain:
            remain[x] = x
        remain[x], remain[n-1-i] = remain[n-1-i], remain[x]
        result.add(remain.pop(n-1-i))
    return result
```
